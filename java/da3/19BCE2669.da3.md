                }
            }
        }
        return trains;
    }

    // sort by time
    private Train[] sortByTime(Train[] trains) {
        for (int i = 0; i < trains.length; i++) {
            for (int j = i + 1; j < trains.length; j++) {
                if (trains[i].time > trains[j].time) {
                    Train temp = trains[j];
                    trains[j] = trains[i];
                    trains[i] = temp;
                }
            }
        }
        return trains;
    }
}
